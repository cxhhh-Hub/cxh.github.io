this指向问题

1,任何情况下直接在script中写入的this都是window

console.log(this)

2,函数中的this

如果非严格模式this指向window，严格模式时，this指向undefined

function fn(){
    console.log(this);
}
fn();

3,箭头函数

不管是否严格模式，this都指向箭头函数外上下文环境的this指向

var fn=()=>{
    console.log(this);
}
fn();

4,对象中的this

对象属性的this 指向对象外上下文环境的this

//this--->window
var a=10;
var obj={
    a:this.a
}
console.log(obj.a)

对象方法(普通函数)中的this，指向当前对象(谁执行该方法，this就指向谁)

var  obj={
    a:1,
    b:function(){
        console.log(this);
    }
}

这里的this与第三条箭头函数中的this指向相同，指向对象外上下文环境中this

var obj={
    a:1,
    b:()=>{
        // this--->window
        console.log(this)
    }
}

var o={
    a:1,
    b:function(){
        console.log(this===o1);//true
        console.log(this===o);//false
    }
}


var o1=Object.create(o);
o1.b();

var o={
    a:1,
    b:function(){
        console.log(this) //o
        var fn=()=>{
            // 3-->箭头函数
            console.log(this);// o
        }
        fn();

        function fns(){
            // 2-->函数指向
            console.log(this);//window
        }
        fns();
        this.d();
    },
    c:()=>{
        // 3-->箭头函数
        console.log(this);//window
    },
    d:function(){
        console.log(this)//调用环境 o
    }
}

o.b();
o.c();

5,回调函数中的this

1,setTimeout，setInterval回调函数

不管是否是严格模式都会指向window

function fn(){
    console.log(this)//window
}

setTimeout(fn,500);

2,通过别的函数内执行当前回调函数

这种通过别的函数内执行当前回调函数中 等同于2–>函数

function fn(){
    console.log(this)//window
}

function fn1(f){
    f();
}
fn1(fn);

fn();
function fn(){
    requestAnimationFrame(fn);
    console.log(this);//window
}

3,递归函数

等同于函数

var i=0;
function fn(){
    console.log(this)//window
    i++;
    if(i>3) return;
    fn();
}
fn();

var obj={
    a:1,
    b:function(){
        console.log(this);//window
    },
    c:function(f){
        f();
    }
}

obj.c(obj.b);

3,默认状态数组部分遍历方法中回调函数中的this

等同于函数     数组的部分遍历方法中如果在最后一个参数中给入内容，那么这个函数中this将会被指向这个给入的内容(有thisArg参数)

var arr=[1,2,3,4];
arr.forEach(function(){
    console.log(this);//window
})

​ arr.filter()

​ arr.forEach()

​ arr.every();

​ arr.find();

​ arr.findIndex();

​ arr.map();

​ arr.some();

上面的这些方法都带有thisArg 所以情况和与下面相同

var arr=[1,2,3,4];
arr.forEach(function(){
    console.log(this)//{a:1}
},{a:1})

var arr=[1,2,3,4];
arr.filter(function(item,index,arr){
    console.log(this)//{a:1}
},{a:1})

var arr=[1,2,3,4];
arr.flatMap(function(item,index,arr){
    console.log(this)//{a:1}
},{a:1})

4,回调函数时的arguments对应项执行回调

回调函数时，如果使用arguments对应项执行回调，那么在被指向的函数中this指向回调该函数的那个函数中上下文环境中arguments

function fn(){
    console.log(this)//fn1的arguments
}

function fn1(){
    arguments[0]()
}
fn1(fn);

5,事件中的回调函数

this指向事件侦听的对象(e.currentTarget)

document.addEventListener("click",clickHandler);

function clickHandler(e){
    console.log(this)
}

document.addEventListener("click",clickHandler);

function clickHandler(e){
    setTimeout(function(){
        console.log(this);
    },0)
}

6,递归中的this

var i=0;
function fn(){
    console.log(this)
    i++;
    if(i>3) return;
    fn();
}
fn();

7,call，apply，bind方法执行时this的指向

如果call,apply,bind传参时，第一个参数传入的是null或者undefined，非严格模式下指向window，严格模式下，传入什么指向什么；如果第一个参数传入的不是null或者undefined，在这个函数中this将会指向被传入这个参数

function fn(){
    console.log(this)
}
fn.call(null,{a:1});//window
fn.apply(undefined,{a:1});//window

function fn(){
    console.log(this)
}
fn.call({a:1});//{a:1}
fn.apply({a:1});//{a:1}

var obj={a:1,b:2};

var fn=clickhandler.bind(obj);

document.addEventListener("click",fn);

function clickhandler(e){
    console.log(this);//{a:1.b:2}
}

8,ES6的类中this的指向

构造函数中的this指向实例了当前类所产生的新的实例对象

类中实例化方法中this指向谁执行该方法，this指向谁

类中静态方法中this指向该类或者该类的构造函数

类中实例化箭头方法，this仍然指向当前类实例化的实例对象

class Box{
    constructor(){
        console.log(this)
    }
    play(){
        console.log(this)
    }
    static run(){
        console.log(this)
    }
    a=()=>{
        console.log(this);
    }
}

var b=new Box();//实例对象
b.play();//实例对象
b.a();//实例对象
Box.run();//该类

9,ES5的原型对象中this的指向

在原型的方法中，this指向实例化当前构造函数的实例化对象（谁执行该方法，this指向谁）

function Box(){

}
Box.prototype.play=function(){
    console.log(this);//实例对象
}
//等同于箭头函数  
Box.prototype.run=()=>{
    console.log(this);//window
}

var b=new Box();
b.play();
b.run();
